;==============================
;========== TEST AIM ==========
;==============================
; 
; To establish if the "place-randomly" procedure operates correctly when there is only 1 turtle in the
; environment.
;
;======================================
;========== TEST DESCRIPTION ==========
;====================================== 
;
; - A turtle is created and its initial coordinates are set to a known location.
; - Turtle just created invokes "place-randomly" until it is not located on its original coordinates.
;
;=====================================
;========== TESTS PERFORMED ==========
;=====================================
;
; - The coordinates that the turtle is finally located on are not equal to its original coordinates.
;
<test>
	create-chrest-turtles 1[ 
		setxy (0) (0)
		place-randomly 
		while [ [pxcor] of patch-here = 0 and [pycor] of patch-here = 0][
			place-randomly
		]
	
		check-equal 
			( [pxcor] of patch-here = 0 and [pycor] of patch-here = 0 ) 
			(false) 
			("when checking the location of the turtle")
	]
</test>
