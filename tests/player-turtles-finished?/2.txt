;==============================
;========== TEST AIM ==========
;==============================
;
; To determine if the "player-turtles-finished?" procedure evaluates to false as expected.
;
;======================================
;========== TEST DESCRIPTION ==========
;====================================== 
;
; - One turtle of each breed in the global 'breeds' list are created.
; - All turtles are asked to set their 'hidden' turtle variable to false.
;
;=====================================
;========== TESTS PERFORMED ==========
;=====================================
;
; - The "player-turtles-finished?"" procedure reports boolean false after invocation.
;
<test>
	foreach(breeds)[
		run (word "create-" ? " 1")
	]

	ask turtles [
		set hidden? (false)
	]

	check-equal 
		(player-turtles-finished?) 
		(false)
		("when checking the output of the 'player-turtles-finished?' procedure")
</test>
